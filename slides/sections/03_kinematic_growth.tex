\begin{frame}[t]{Multiplicative Decomposition --- Growth vs.\ Elasticity}
  Kinematic growth splits the deformation into a stress-free \textbf{growth} part
  $\bm{F}_g$ (how the tissue adds mass) and an \textbf{elastic} part $\bm{F}_e$ ---
  the only part that carries stress.
  \dlab{general form --- a product of tensors}
  \giveneq{\bm{F}=\bm{F}_e\,\bm{F}_g,\qquad\text{only }\bm{F}_e\text{ stores energy.}}
  \zlab{reduction used here --- scalar stretches}
  \slboxeq{2.2em}{eq:split}{\lambda = \lambda_e \,\lambda_g}
  Growth is \emph{one} inelastic stretch $\lambda_g$; the elastic stretch
  $\lambda_e=\lambda/\lambda_g$ is what the constituent laws see. (Constrained
  mixtures will later give each cohort its own natural configuration.)
\end{frame}

\begin{frame}[t]{The Idea, and the Kinematics}
  Postulate a stress-free \textbf{growth} deformation $\bm{F}_g$; only the elastic
  part $\bm{F}_e$ carries load. Grown pieces need not fit together ---
  $\bm{F}_g$ leaves an \emph{incompatible} stress-free state; $\bm{F}_e$ forces
  them back into one body (residual stress).
  \vfill
  \begin{center}\kinmap[1.0]\end{center}
  \vfill
\end{frame}

\begin{frame}[t]{The Growth Tensor Must Be Prescribed}
  $\bm{F}_g$ is \textbf{chosen by the modeler}, not derived --- it encodes
  \emph{how} the tissue is assumed to add mass. A common choice grows the tissue
  along a single direction $\bm{e}_f$ (a wall-thickening or \emph{fiber}
  direction):
  \dlab{prescribed growth tensor --- this is an \emph{input}}
  \slboxeq{2.6em}{eq:growthtensor}{\bm{F}_g = \bm{I} + (\theta-1)\,
    \bm{e}_f\!\otimes\bm{e}_f
    \;=\;\mathrm{diag}(1,\,1,\,\theta)}
  \zlab{the one scalar that actually evolves}
  \giveneq{\theta = \frac{M}{M_0}\ \ (\text{mass / volume ratio}),\qquad
    \det\bm{F}_g=\theta.}
  Here $\bm{e}_f$ is the growth direction --- it could be the radial
  (wall-thickening) direction, or a fiber direction. Everything about the growth
  \emph{pattern} is fixed a priori; only the scalar \textbf{growth stretch}
  $\theta$ evolves in time.
  \srccite{Rodriguez, Hoger, McCulloch, J Biomech (1994)}
\end{frame}

\begin{frame}[t]{The Growth Law}
  The growth \emph{pattern} $\bm{F}_g$ is fixed; the growth law evolves its single
  scalar $\theta$ to null the stress deviation from the set-point (first-order,
  production $\propto$ current mass).
  \dlab{general form --- rate of the growth tensor via its scalar}
  \giveneq{\dot{\bm{F}}_g = \dot\theta\,\bm{e}_f\!\otimes\bm{e}_f,\qquad
    \dot\theta = k_g\,\theta\left(\tfrac{\bar\sigma}{\bar\sigma_h}-1\right).}
  \zlab{scalar growth law --- the only unknown in $\bm{F}_g$}
  \slboxeq{2.6em}{eq:kingrowth}{\frac{\dd\theta}{\dd t}
    = k_g\,\theta\left(\frac{\bar\sigma}{\bar\sigma_h}-1\right)}
  The set-point $\bar\sigma_h$ is \textbf{prescribed}: wherever growth stops
  ($\dot\theta=0$), stress is homeostatic \emph{by construction}. At each instant
  $\lambda$ follows from equilibrium
  $\bar\sigma(\lambda)=\sigma_{\text{req}}(\lambda,\theta)$.
  \srccite{Rodriguez, Hoger, McCulloch, J Biomech (1994)}
\end{frame}

\begin{frame}[t]{Stability Is Imposed, Not Predicted}
  Is the homeostatic state $\theta^*$ stable? Perturb it, $\theta=\theta^*+\delta$,
  and linearise the growth law \eqref{eq:kingrowth} about $\theta^*$ (where
  $\bar\sigma=\bar\sigma_h$):
  \giveneq{\dot\delta = \underbrace{k_g\,\theta^*\,
    \frac{1}{\bar\sigma_h}\left.\frac{\dd\bar\sigma}{\dd\theta}\right|_{\theta^*}}_{\displaystyle =:\,\alpha}\;\delta.}
  The perturbation decays ($\delta\to0$, \textbf{stable}) iff $\alpha<0$, i.e.
  \slboxeq{2.2em}{eq:stabkin}{\left.\frac{\dd\bar\sigma}{\dd\theta}\right|_{\theta^*}<0
    \quad\Longleftrightarrow\quad\text{adding mass \emph{lowers} the stress it must carry.}}
  For the intact artery this holds, but a strong enough insult flips the sign
  (Laplace $\bar\sigma\propto\lambda^2$) --- then $\alpha>0$ and growth runs away.
  The set-point is \emph{prescribed}, so the model can only \emph{impose}
  stability, never \emph{predict} it.
\end{frame}

\begin{frame}[t]{The \texttt{gr} Toolbox --- Shared Setup for Every Coding Exercise}
  \footnotesize
  Every coding exercise reuses \emph{one} problem definition, so the four theories
  are compared on identical numbers. A \texttt{Model} bundles the constituents with
  the homeostatic pressure $P_h=13.3$~kPa ($\sim$100~mmHg) and reference mid-wall
  radius $R=1$~mm; the reference thickness follows from the Laplace balance
  $H=P_hR/\bar\sigma_h$ \eqref{eq:laplace}.
  \vspace{-0.2em}
  \begin{columns}[T]
    \begin{column}{0.56\textwidth}
      \textbf{Default arterial constituents} (mass fraction $\phi_0$, deposition
      stretch $G$, law, turnover rate $k_d$, gain $K_\sigma$):
      \renewcommand{\arraystretch}{1.15}
      \begin{tabular}{@{}lccl@{}}
        \toprule
        & $\phi_0$ & $G$ & law, $k_d$\,[1/d], $K_\sigma$\\
        \midrule
        elastin  & 0.30 & 1.40 & neo-Hooke $c{=}90$; $k_d{=}0$\\
        collagen & 0.35 & 1.08 & Fung $c_1{=}250,c_2{=}8$; $\tfrac1{20}$; 1\\
        smc      & 0.35 & 1.20 & Fung $c_1{=}120,c_2{=}3$; $\tfrac1{20}$; 1\\
        \bottomrule
      \end{tabular}
      \par\smallskip
      Set-point per constituent $\sigma_h^k=\sigma^k(G^k)$; mixture
      $\bar\sigma_h=\sum_k\phi_0^k\sigma_h^k$. Elastin never turns over
      ($k_d^e=0$) and is the only \emph{degradable} constituent (aneurysm).
    \end{column}
    \begin{column}{0.44\textwidth}
      \textbf{Geometry \& equilibrium.} At every step the supplied mixture stress
      must equal the Laplace required stress
      \giveneq{\bar\sigma(\lambda)=\bar\sigma_h\,\gamma\,\frac{\lambda^{2}}{m},}
      solved for $\lambda$ by a bracketed root find on $[0.2,20]$; \emph{no root}
      $\Rightarrow$ runaway. Here $m=M/M_0$ is the mass ratio, $\gamma$ the load
      factor, and the exponent 2 is the Laplace term $\bar\sigma=Pa/h$.
      \par\smallskip
      \textbf{Insult.} \texttt{Insult} ramps a step from \texttt{t\_on} over
      \texttt{ramp} days: \texttt{pressure\_factor} $\gamma$ (hypertension) or
      \texttt{elastin\_surviving} $s$ (aneurysm).
    \end{column}
  \end{columns}
  \vspace{-0.1em}
  \textbf{Running an exercise.} Each exercise is a small YAML \emph{input file};
  one runner drives them all --- no Python to edit:
  \scodeline{uv run python run.py configs/exNN\_*.yaml}
  The file picks the \texttt{theory}, \texttt{geometry}, \texttt{insult} and
  parameters; the run prints $\bar\sigma/\bar\sigma_h$, mass and \texttt{diverged}
  and plots the history. For a \textbf{parametric study}, list values in the
  file's \texttt{sweep:} block or on the command line:
  \scodeline{... --sweep insult.pressure\_factor=1.2,1.5,2.0 \ \ \# one curve per value}
  \scodeline{... --set\ \ \ model.constituents.collagen.gain=3.0 \ \# change any one value}
\end{frame}

    \sprob{A pressurised artery is made \emph{hypertensive} by a sustained
    $+50\%$ pressure step. Does the wall thicken back to its \emph{prescribed}
    homeostatic stress \eqref{eq:kingrowth}, or run away once the pressure step is
    large enough?}
    \srun{\scodeline{uv run python run.py configs/ex02\_kinematic\_growth.yaml}

      climbs to the pressure factor.}
    \sstudy{Overlay several growth rates (the file's \texttt{sweep:} block), or
      push the pressure until it diverges:
      \scodeline{... --sweep simulate.k\_g=0.02,0.05,0.2\ \ \ \ \ \# vary the gain}
      \scodeline{... --set insult.pressure\_factor=3.0\ \ \ \ \ \ \ \# harder insult}}
  \end{setupbox}
\end{frame}
