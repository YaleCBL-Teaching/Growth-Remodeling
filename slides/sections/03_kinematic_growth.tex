\begin{frame}[t]{Multiplicative Decomposition: Growth vs.\ Elasticity}
  Split the deformation into a stress-free \textbf{growth} part $\bm{F}_g$ and a stress-carying \textbf{elastic} part $\bm{F}_e$.
  \dlab{a product of tensors}
  \giveneq{\bm{F}=\bm{F}_e\,\bm{F}_g,\qquad\text{only }\bm{F}_e\text{ stores energy.}}
  \zlab{scalar stretches}
  \slboxeq{eq:split}{\lambda = \lambda_e \,\lambda_g}
  \srccite{Rodriguez, Hoger, McCulloch, J Biomech (1994)}
\end{frame}

\begin{frame}[t]{Multiplicative Decomposition: Growth vs.\ Elasticity}
  $\bm{F}_g$ yields an \emph{incompatible} stress-free state, $\bm{F}_e$ restores compability via residual stress.
  \vfill
  \begin{center}\kinmap[1.0]\end{center}
  \vfill
\end{frame}

\begin{frame}[t]{Kinematic Growth Tensor}
  $\bm{F}_g$ encodes \emph{how} the tissue is assumed to add mass. A common choice grows the tissue
  an amount $\theta$ along a fixed direction $\bm{f}$:
  \dlab{prescribed growth tensor}
  \slboxeq{eq:growthtensor}{\bm{F}_g = \bm{I} + (\theta-1)\,
    \bm{f}\!\otimes\bm{f}
    \;=\;\mathrm{diag}(1,\,1,\,\theta)}
  \zlab{growth stretch}
  \giveneq{\lambda_g\ \ (\text{scalar growth stretch}),
    \qquad \theta=\det\bm{F}_g=M/M_0.}
  \srccite{Rodriguez, Hoger, McCulloch, J Biomech (1994)}
\end{frame}

\begin{frame}[t]{Growth Law}
  The growth \emph{pattern} $\bm{F}_g$ is typically fixed. Its rate
  $\dot{\bm{F}}_g=\dot\theta\,\bm{f}\!\otimes\bm{f}$ reduces to \emph{one}
  scalar ODE:
  \slboxeq{eq:kingrowth}{\frac{\dd\theta}{\dd t}
    = k_g\,\theta\left(\frac{\bar\sigma}{\bar\sigma_h}-1\right)}
  Since $k_g\,\theta>0$, the sign of the response follows the stress deviation:
  \giveneq{\frac{\dd\theta}{\dd t}
    \begin{cases}
      >0 & \bar\sigma>\bar\sigma_h \quad\text{(growth)}\\[2pt]
      =0 & \bar\sigma=\bar\sigma_h \quad\text{(homeostasis)}\\[2pt]
      <0 & \bar\sigma<\bar\sigma_h \quad\text{(atrophy)}
    \end{cases}}
  \srccite{Rodriguez, Hoger, McCulloch, J Biomech (1994)}
\end{frame}

\begin{frame}[t]{Getting Started with the Code}
  \small
  One runner drives \emph{every} exercise: you never edit Python, just point
  it at a small input file. \textbf{First-time setup} (needs Python $\ge3.10$;
  dependencies: NumPy, SciPy, Matplotlib, PyYAML):
  \par\smallskip
  \scodeline{git clone https://github.com/YaleCBL-Teaching/Growth-Remodeling}
  \scodeline{cd Growth-Remodeling}
  \vspace{0.2em}
  \begin{columns}[T]
    \begin{column}{0.4\textwidth}
      \textbf{\color{YaleBlue}uv} (recommended):
      \scodeline{uv sync}
    \end{column}
    \begin{column}{0.4\textwidth}
      \textbf{\color{YaleBlue}pip} (your own Python):
      \scodeline{python -m venv .venv}
      \scodeline{pip install -e .}
    \end{column}
  \end{columns}
  \vspace{0.5em}
  \textbf{What is where:}
  \begin{itemize}\setlength\itemsep{1pt}
    \item \texttt{run.py}: the single runner for all exercises
    \item \texttt{configs/}: one YAML input file per exercise \emph{(you edit these)}
    \item \texttt{src/gr/}: the models \& maths (read, don't edit)
    \item \texttt{docs/}: documentation
    \item \texttt{slides/}: semianr notes
  \end{itemize}
  \srccite{github.com/YaleCBL-Teaching/Growth-Remodeling}
\end{frame}

    \sprob{A pressurised artery is made \emph{hypertensive} by a sustained
    pressure step. Kinematic growth thickens the wall. Does the wall stress
    $\bar\sigma/\bar\sigma_h$ return to the \emph{prescribed} set-point?}
  \end{setupbox}

\begin{frame}[t]{Running an Exercise}
  \small
  Point the runner at the input file (\texttt{uv run}, or plain \texttt{python}
  once the venv is activated):
  \par\smallskip
  \scodeline{uv run python run.py configs/ex01\_kinematic\_growth.yaml}
  \vspace{0.3em}\par
  \textbf{You get:} the key numbers ($\bar\sigma/\bar\sigma_h$, mass,
  \texttt{diverged}) print to the terminal, and the time history opens as a plot.
  \vspace{0.5em}\par
  \textbf{Change one value:} edit the YAML, or override on the command line:
  \par\smallskip
  \scodeline{uv run python run.py configs/ex01\_kinematic\_growth.yaml --set insult.pressure\_factor=2}
  \vspace{0.3em}\par
  \textbf{Parametric study:} one curve per value:
  \par\smallskip
  \scodeline{... --sweep insult.pressure\_factor=1.5,2,3}
\end{frame}

    pressure steps. Does the outcome ever change?}
    \qhint{$\bar\sigma/\bar\sigma_h$ settles at 1 while $\theta$ climbs to the

    needs the mixture / aneurysm story (later).}

\begin{frame}[t]{}
  \begin{answerbox}{Kinematic growth}
    The stress plateau \emph{always} lands on the prescribed set-point. The wall
    simply thickens more for a larger pressure step. Kinematic growth
    \emph{imposes} stability; it cannot predict runaway.
  \end{answerbox}
