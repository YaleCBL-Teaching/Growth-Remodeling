\begin{frame}[t]{Multiplicative Decomposition --- Growth vs.\ Elasticity}
  Kinematic growth splits the deformation into a stress-free \textbf{growth} part
  $\bm{F}_g$ (how the tissue adds mass) and an \textbf{elastic} part $\bm{F}_e$ ---
  the only part that carries stress.
  \dlab{general form --- a product of tensors}
  \giveneq{\bm{F}=\bm{F}_e\,\bm{F}_g,\qquad\text{only }\bm{F}_e\text{ stores energy.}}
  \zlab{reduction used here --- scalar stretches}
  \slboxeq{2.2em}{eq:split}{\lambda = \lambda_e \,\lambda_g}
  Growth is \emph{one} inelastic stretch $\lambda_g$; the elastic stretch
  $\lambda_e=\lambda/\lambda_g$ is the only part the material law sees. Here the
  wall is treated as a \emph{single} material --- no separate constituents yet.
  \srccite{Rodriguez, Hoger, McCulloch, J Biomech (1994)}
\end{frame}

\begin{frame}[t]{The Idea, and the Kinematics}
  Postulate a stress-free \textbf{growth} deformation $\bm{F}_g$; only the elastic
  part $\bm{F}_e$ carries load. Grown pieces need not fit together ---
  $\bm{F}_g$ leaves an \emph{incompatible} stress-free state; $\bm{F}_e$ forces
  them back into one body (residual stress).
  \vfill
  \begin{center}\kinmap[1.0]\end{center}
  \vfill
\end{frame}

\begin{frame}[t]{The Growth Tensor Must Be Prescribed}
  $\bm{F}_g$ is \textbf{chosen by the modeler}, not derived --- it encodes
  \emph{how} the tissue is assumed to add mass. A common choice grows the tissue
  along a single direction $\bm{e}_f$ (a wall-thickening or \emph{fiber}
  direction):
  \dlab{prescribed growth tensor --- this is an \emph{input}}
  \slboxeq{2.6em}{eq:growthtensor}{\bm{F}_g = \bm{I} + (\theta-1)\,
    \bm{e}_f\!\otimes\bm{e}_f
    \;=\;\mathrm{diag}(1,\,1,\,\theta)}
  \zlab{0D-specific reduction --- the growth tensor becomes one stretch}
  \giveneq{\bm{F}_g\ \longrightarrow\ \lambda_g\ \ (\text{scalar growth stretch}),
    \qquad \theta=\det\bm{F}_g=M/M_0.}
  Here $\bm{e}_f$ is the growth direction (e.g.\ wall-thickening). The growth
  \emph{pattern} is fixed a priori; only the scalar \textbf{growth stretch}
  $\lambda_g$ (equivalently the mass ratio $\theta$) evolves in time.
  \srccite{Rodriguez, Hoger, McCulloch, J Biomech (1994)}
\end{frame}

\begin{frame}[t]{The Growth Law}
  The growth \emph{pattern} $\bm{F}_g$ is fixed; the growth law evolves its single
  scalar to null the stress deviation from the set-point (first-order, production
  $\propto$ current mass). Since $\bm{F}_g$ is fixed by $\theta$, its rate
  $\dot{\bm{F}}_g=\dot\theta\,\bm{e}_f\!\otimes\bm{e}_f$ reduces to \emph{one}
  scalar ODE --- the \emph{same} law in the full 3D tensor model and the 0D
  reduction (both track the scalar $\theta$):
  \slboxeq{2.6em}{eq:kingrowth}{\frac{\dd\theta}{\dd t}
    = k_g\,\theta\left(\frac{\bar\sigma}{\bar\sigma_h}-1\right)}
  The set-point $\bar\sigma_h$ is \textbf{prescribed}: wherever growth stops
  ($\dot\theta=0$), stress is homeostatic \emph{by construction}. So the model can
  only \emph{impose} its set-point, never \emph{predict} stability --- that is
  what the mixture theories add.
  \srccite{Rodriguez, Hoger, McCulloch, J Biomech (1994)}
\end{frame}

% ---- problem setup: ex01 ---------------------------------------------------
\begin{frame}[t]{}
  \begin{setupbox}{Example --- hypertensive artery (\texttt{ex01})}
    \sprob{A pressurised artery is made \emph{hypertensive} by a sustained
    pressure step. Kinematic growth thickens the wall --- does the wall stress
    $\bar\sigma/\bar\sigma_h$ return to the \emph{prescribed} set-point?}
  \end{setupbox}
\end{frame}

% ---- how to run (concise), placed AFTER the problem setup ------------------
\begin{frame}[t]{Running the Exercises}
  One runner drives every exercise --- no Python to edit. Run the input file:
  \scodeline{uv run python run.py configs/ex01\_kinematic\_growth.yaml}
  \textbf{Results:} the key numbers ($\bar\sigma/\bar\sigma_h$, mass,
  \texttt{diverged}) print to the terminal, and the time history opens as a plot.
  \vspace{0.7em}\par
  \textbf{Parametric study} --- one curve per value, from the command line:
  \scodeline{... --sweep insult.pressure\_factor=1.5,2,3}
  Every parameter in the file can be swept or overridden the same way.
\end{frame}

\begin{frame}[t]{}
  \begin{answerbox}{Kinematic growth}
    The stress plateau \emph{always} lands on the prescribed set-point --- the wall
    simply thickens more for a larger pressure step. Kinematic growth
    \emph{imposes} stability; it cannot predict runaway.
  \end{answerbox}
