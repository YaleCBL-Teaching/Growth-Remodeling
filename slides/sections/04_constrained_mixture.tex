% ---- the idea: track every cohort (start of CMM) ---------------------------
\begin{frame}[t]{The Idea: Track Every Cohort}
  All constituents are \textbf{constrained to deform together}, but each carries
  its \emph{own} natural configuration and is continuously \textbf{produced and
  removed} --- so the tissue holds a whole \emph{history} of cohorts.
  \vfill
  \begin{center}\humphreygr[0.70]\end{center}
  \vspace{-0.1em}
  \begin{center}\footnotesize\color{black!70}
  $\kappa$: configuration \;\textbullet\; $\tau$: deposition time \;\textbullet\;
  $s$: current time \;\textbullet\; $\alpha\!\in\!\{e,c,m\}$: constituent
  \;\textbullet\; $\bm{G}^{\alpha}$: deposition stretch \;\textbullet\;
  $\bm{F}^{\alpha}_{n(\tau)}$: constituent deformation
  \end{center}
  \vfill
  \srccite{Humphrey, J Elasticity (2021)}
\end{frame}

% ---- constituents & mass fractions (moved from biology) + deposition stretch
\begin{frame}[t]{The Cast of Characters: Constituents \& Mass Fractions}
  \footnotesize
  The wall is a mixture of three constituents, each with a reference \textbf{mass
  fraction} $\phi_0^{\alpha}$ ($\sum_\alpha\phi_0^{\alpha}=1$) and a
  \textbf{deposition stretch} $G^{\alpha}$ --- the prestretch cells lay it down at,
  which \emph{defines} its homeostatic stress
  $\sigma_h^{\alpha}=\sigma^{\alpha}(G^{\alpha})$; the mixture set-point is
  $\bar\sigma_h=\sum_\alpha\phi_0^{\alpha}\sigma_h^{\alpha}$.
  \vspace{0.3em}
  \renewcommand{\arraystretch}{1.3}
  \begin{center}
  \begin{tabular}{@{}p{0.15\textwidth}p{0.28\textwidth}ccp{0.24\textwidth}@{}}
    \toprule
    \textbf{Constituent} & \textbf{Role} & $\phi_0$ & $G^{\alpha}$ & \textbf{law; turnover $1/k_d$}\\
    \midrule
    \textbf{\color{YaleBlue}Elastin} & bears load at low stretch & 0.30 & 1.40 &
      neo-Hooke; \textbf{none} (decades)\\
    \textbf{\color{YaleBlue}Collagen} & stiff, crimped; anti-rupture & 0.35 & 1.08 &
      Fung; fast ($\sim$20\,d)\\
    \textbf{\color{YaleBlue}Smooth muscle} & passive + active; the sensor & 0.35 & 1.20 &
      Fung; fast ($\sim$20\,d)\\
    \bottomrule
  \end{tabular}
  \end{center}
  \vspace{0.3em}
  \textbf{Elastin cannot be renewed} ($k_d^e=0$) --- its loss drives
  \emph{aneurysm}; collagen \& SMC \textbf{turn over}, so the wall can
  \emph{adapt}.
  \srccite{Humphrey, Eberth, Dye, Gleason, J Biomech (2009)}
\end{frame}

% ---- why "constrained": one shared tissue stretch --------------------------
\begin{frame}[t]{Why ``Constrained''? Every Cohort Shares One Stretch}
  A cohort of $\alpha$ deposited at $\tau$ is laid down at $G^{\alpha}$ and then
  stretches \emph{with the mixture}: every constituent and cohort is
  \textbf{constrained} to the same tissue stretch $\lambda(t)$.
  \dlab{general form --- constituent-specific deformation}
  \giveneq{\bm{F}^{\alpha}_{n(\tau)}(t) = \bm{F}(t)\,\bm{F}^{-1}(\tau)\,\bm{G}^{\alpha}.}
  \zlab{0D reduction --- scalar elastic stretch}
  \slboxeq{2.2em}{eq:cohort}{\lambda_e^{\alpha}(t,\tau) = G^{\alpha}\,\frac{\lambda(t)}{\lambda(\tau)}}
  The shared $\lambda(t)$ is the \emph{constraint}. At birth ($t=\tau$) the cohort
  sits at $G^{\alpha}$ (hence $\sigma_h^{\alpha}$); older cohorts measure the same
  $\lambda$ from an earlier $\lambda(\tau)$.
  \srccite{Humphrey, Rajagopal, Math Models Methods Appl Sci (2002)}
\end{frame}

\begin{frame}[t]{Mixture Stress = Mass-Weighted Sum}
  Each constituent contributes in proportion to its mass --- the \textbf{rule of
  mixtures}, summed over all surviving cohorts (all evaluated at the \emph{shared}
  $\lambda$):
  \slboxeq{3.2em}{eq:mixstress}{\bar\sigma(t) = \frac{1}{M_{\text{tot}}}\sum_\alpha\!\left[
     M^{\alpha}(0)\,q^{\alpha}(t,0)\,\sigma^{\alpha}\!\big(G^{\alpha}\lambda(t)\big)
     + \int_0^t\! m^{\alpha}(\tau)\,q^{\alpha}(t,\tau)\,\sigma^{\alpha}\!\big(\lambda_e^{\alpha}\big)\,\dd\tau\right]}
  Each step evaluates this sum, then solves
  $\bar\sigma(\lambda)=\sigma_{\text{req}}(\lambda)$ for the current stretch. The
  surviving fraction $q^{\alpha}$ and production rate $m^{\alpha}$ come from the mass
  balance next.
  \srccite{Humphrey, Rajagopal, Math Models Methods Appl Sci (2002)}
\end{frame}

\begin{frame}[t]{Mass Balance --- The Heredity Integral}
  Each turnover constituent's mass is the surviving initial material plus all it
  has produced since, aged by a survival function:
  \slboxeq{2.6em}{eq:heredity}{M^{\alpha}(t) = M^{\alpha}(0)\,q^{\alpha}(t,0)
    + \int_0^t m^{\alpha}(\tau)\,q^{\alpha}(t,\tau)\,\dd\tau}
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \textbf{Survival} (Poisson removal, rate $k_d^{\alpha}$):
      $\;q^{\alpha}(t,\tau)=e^{-k_d^{\alpha}(t-\tau)}$, mean lifetime $1/k_d^{\alpha}$
      (elastin: $k_d^e=0$).
    \end{column}
    \begin{column}{0.5\textwidth}
      \textbf{Production} $\propto$ mass, driven by stress:
      $\;m^{\alpha}=k_d^{\alpha}M^{\alpha}\,\Upsilon$,
      $\;\Upsilon=1+K_\sigma^{\alpha}\!\left(\tfrac{\bar\sigma}{\bar\sigma_h}-1\right)$.
    \end{column}
  \end{columns}
  \srccite{Humphrey, Rajagopal, Math Models Methods Appl Sci (2002)}
\end{frame}

% ---- quick question: computational cost of the heredity integral -----------
\begin{frame}[t]{}
  \begin{yourturnbox}{Quick Check --- How Expensive Is This?}
    \qq{At each of $N$ time steps the mixture stress sums over \emph{all} cohorts
    deposited so far. How does the total cost scale with $N$?}
    \qhint{Step $k$ re-sums $\sim k$ stored cohorts; add up $\sum_{k=1}^{N}k$.}
  \end{yourturnbox}
  \vfill
  \begin{center}
  $\displaystyle\sum_{k=1}^{N} k=\frac{N(N+1)}{2}=O(N^2)$
  \end{center}
  \vfill
  \begin{center}\small
  The full model stores and re-sums the \emph{entire} history --- $O(N^2)$ work.
  Removing that cost is exactly what the \textbf{homogenized} model does next.
  \end{center}
\end{frame}
