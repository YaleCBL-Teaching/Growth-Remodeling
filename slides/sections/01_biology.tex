\begin{frame}[t]{}
  \vfill\begin{center}{\Huge\color{YaleBlue} 1.\ \ The biology}\\[0.5em]
  {\color{BoxGray} why living tissues grow and remodel}\end{center}\vfill
\end{frame}

\begin{frame}[t]{Living materials are never ``finished''}
  A steel beam is made once and wears out. A living artery is
  \textbf{continuously torn down and rebuilt}: cells deposit and degrade
  structural proteins while sensing the loads they carry.
  \begin{itemize}
    \item \textbf{Growth} --- change in \emph{mass}.
    \item \textbf{Remodeling} --- change in \emph{microstructure} (fiber
    orientation, natural length, composition).
  \end{itemize}
  \vfill
  \begin{center}\color{AccentRed}
  Will a hypertensive artery thicken and stabilise, or an aneurysm dilate to
  rupture?
  \end{center}
\end{frame}

\begin{frame}[t]{The cast of characters in an arterial wall}
  \small
  \begin{center}
  \renewcommand{\arraystretch}{1.4}
  \begin{tabular}{@{}p{0.17\textwidth}p{0.34\textwidth}p{0.36\textwidth}@{}}
    \toprule
    \textbf{Constituent} & \textbf{Mechanical role} & \textbf{Turnover in adults}\\
    \midrule
    \textbf{\color{YaleBlue}Elastin} & soft; bears load at low stretch &
      \textbf{none} --- half-life of decades \\
    \textbf{\color{YaleBlue}Collagen} & stiff, crimped; guards against rupture &
      \textbf{fast} --- weeks to months \\
    \textbf{\color{YaleBlue}Smooth muscle} & passive load + active contraction;
      the sensor cell & fast; switches phenotype \\
    \bottomrule
  \end{tabular}
  \end{center}
  \vfill
  \begin{itemize}\small
    \item \textbf{Elastin cannot be renewed} --- its loss triggers \emph{aneurysm}.
    \item \textbf{Collagen \& SMC turn over constantly} --- so the wall can
    \emph{adapt}, or \emph{run away}.
  \end{itemize}
  \srccite{Humphrey, Eberth, Dye, Gleason, J Biomech (2009)}
\end{frame}

\begin{frame}[t]{Deposition pre-stretch}
  Cells cross-link each new fiber \emph{while the wall is loaded}, so it is
  \textbf{born pre-stretched} to the same target state. Each cohort keeps its
  \emph{own} natural configuration --- a {\color{YaleBlue}\textbf{constrained
  mixture}}: deforming together now, each remembering a different stress-free state.
  \vfill
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \begin{center}\textbf{\color{BoxGray} no deposition pre-stress}\end{center}
      New material enters slack ($\sigma_{\text{pre}}=0$); as old fibers are
      removed, the replacement carries no load --- the homeostatic state
      \textbf{cannot be maintained}.
    \end{column}
    \begin{column}{0.5\textwidth}
      \begin{center}\textbf{\color{YaleBlue} with deposition pre-stress}\end{center}
      New material enters at the homeostatic prestress
      $\sigma_{\text{pre}}=\sigma_h$ (stretch $G^k$); each replacement immediately
      bears its share --- the state is \textbf{maintained through turnover}.
    \end{column}
  \end{columns}
  \vfill
  \begin{center}\color{AccentRed}
  This one convention is what lets a continuously-renewed tissue hold a steady
  mechanical state.
  \end{center}
\end{frame}

\begin{frame}[t]{Cells chase a mechanical set-point}
  Cells regulate a \textbf{target (homeostatic) stress}: above it they build more,
  below it removal wins --- a negative feedback that thickens a hypertensive wall
  until stress returns to normal.
  \bigfigw{fut_homeostasis_feedback.png}{0.86\linewidth}{Regulated variables are
  compared to set-points; the mismatch drives constituent-specific G\&R.}
  \figcredit{Adapted from Humphrey, J Elasticity (2021)}
\end{frame}

\begin{frame}[t]{Fundamental observation: a stress set-point}
  Wolinsky \& Glagov (1967): Cauchy stress in the aortic media is nearly
  \textbf{constant across mammals} spanning orders of magnitude in body mass ---
  direct evidence of a mechanical set-point. (Davis 1867 \& Wolff 1892: tissue
  adds material along sustained tension; Grossman 1975: pressure overload thickens
  the wall just enough to normalise stress.)
  \bigfigw{motiv_wolinsky_glagov.png}{0.55\linewidth}{Wall tension vs.\ body mass
  across species.}
  \figcredit{Fig.: A. Gebauer (TUM lecture); data Wolinsky \& Glagov, Circ Res (1967)}
\end{frame}

\begin{frame}[t]{Two canonical scenarios we will simulate}
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \textbf{\color{YaleBlue}Hypertension} --- sustained pressure step. Expect
      \emph{adaptation}: the wall thickens, stress returns to set-point.
    \end{column}
    \begin{column}{0.5\textwidth}
      \textbf{\color{YaleBlue}Aneurysm} --- irreversible elastin loss. Geometry can
      feed \emph{positively} back on stress: stabilise larger, \emph{or} dilate
      without bound.
    \end{column}
  \end{columns}
  \vfill
  \begin{center}\large\color{AccentRed}
  \emph{When does a tissue adapt, and when does it run away?}
  \end{center}
  \vfill
  {\footnotesize\color{BoxGray} The four theories answer this differently --- and
  the code lets you explore each.}
\end{frame}
