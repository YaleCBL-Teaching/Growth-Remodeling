\begin{frame}[t]{Reference and Current Configuration}
  A body is described by a \textbf{reference} $\mathcal{B}_0$ (points $\bm{X}$) and
  a \textbf{current} $\mathcal{B}_t$ (points $\bm{x}$) configuration, related by the motion
  $\bm{x}=\bm{\varphi}(t,\bm{X})$.

\begin{frame}[t]{The Deformation Gradient}
  The \textbf{deformation gradient} maps reference line elements to current ones.
  \dlab{a $3\times3$ tensor}
  \slboxeq{eq:defgrad}{\dd\bm{x} = \bm{F}\,\dd\bm{X},\qquad
    \bm{F}=\frac{\partial\bm{\varphi}}{\partial\bm{X}},\qquad J=\det\bm{F}}
  $J$ is the volume ratio, \textbf{incompressibility} means $J=1$.
  \zlab{a scalar}
  \giveneq{\bm{F}\ \longrightarrow\ \lambda=\frac{\text{current length}}
    {\text{reference length}},\qquad \lambda=1\ \text{undeformed}.}
\end{frame}

\begin{frame}[t]{Strain: A Rotation-Free Measure of Stretch}
  Removing rigid rotation gives the \textbf{right Cauchy--Green} tensor and the \textbf{Green--Lagrange strain}.
  \dlab{symmetric $3\times3$ tensors}
  \slboxeq{eq:strain}{\bm{C}=\bm{F}^\mathrm{T}\bm{F} = (\bm{R}\bm{U})^\mathrm{T}\bm{R}\bm{U} = \bm{U}^\mathrm{T} \bm{R}^\mathrm{T} \bm{R}^\mathrm{T} \bm{U} = \bm{U}^2,\qquad
    \bm{E}=\tfrac12\!\left(\bm{C}-\bm{I}\right)}
  \zlab{scalars}
  \giveneq{C=\lambda^2,\qquad E=\tfrac12(\lambda^2-1).}
\end{frame}

\begin{frame}[t]{Stress: "Force per Area"}
  \textbf{Cauchy stress}: \emph{current} force per unit \emph{current} area.
  \dlab{spatial symmetric $3\times3$ tensor (from 2nd Piola--Kirchhoff stress $\bm{S}$)}
  \giveneq{\bm{\sigma}=\tfrac1J\,\bm{F}\,\bm{S}\,\bm{F}^\mathrm{T},\qquad
    \bm{S}=2\,\frac{\partial W}{\partial\bm{C}}, \quad W ~ \text{strain energy density function}.}
  \zlab{scalar reduction ($C=\lambda^2$, $J=1$), the $\sigma_\theta$ in Laplace \eqref{eq:laplace}}
  \giveneq{\sigma=\tfrac1J\,\lambda\,S\,\lambda=\lambda^2 S
    =2\lambda^2\,\frac{\partial W}{\partial C}=\lambda\,\frac{\dd W}{\dd\lambda}.}
\end{frame}

\begin{frame}[t]{Hyperelasticity: Stress from a Stored-Energy Function}
  A \textbf{hyperelastic} material stores energy $W$ (deformation is path-independent).
  \dlab{energy $W(\bm{C})$, stress $\bm{S}=2\,\partial W/\partial\bm{C}$}
  \vspace{-0.3em}
  \[
    W_{\text{neo-Hooke}}=\tfrac{c}{2}\,(I_1-3),\qquad
    W_{\text{Fung}}=\tfrac{c_1}{4c_2}\!\left[e^{\,c_2(I_4-1)^2}-1\right]
  \]
  \vspace{-0.6em}\par
  {\footnotesize$I_1=\operatorname{tr}\bm{C}$ (isotropic), \ $I_4=\bm{f}\cdot\bm{C}\,\bm{f}=\lambda_{\text{f}}^{2}$, the squared stretch along the reference fiber direction $\bm{f}$ (anisotropic).}
  \zlab{reduction used here, one stretch $\lambda$, scalar $\sigma=\lambda\,\dd W/\dd\lambda$}
  \vspace{-0.3em}
  \slboxeq{eq:fung}{\sigma_{\text{neo-Hooke}}=c\,(\lambda^2-\lambda^{-1}),\qquad
    \sigma_{\text{Fung}}=c_1\,\lambda^2(\lambda^2-1)\,e^{\,c_2(\lambda^2-1)^2}}
  \srccite{constituent laws as in Latorre, Humphrey, Comput Methods Appl Mech Eng (2020)}
\end{frame}
