\begin{frame}[t]{Reference and Current Configuration}
  A body is described in two configurations: the \textbf{reference}
  $\mathcal{B}_0$ (material points $\bm{X}$) and the \textbf{current}
  $\mathcal{B}_t$ (spatial points $\bm{x}$). The \textbf{motion}
  $\bm{x}=\bm{\varphi}(t,\bm{X})$ carries one to the other.
  \vfill
  \begin{center}\configmap[1.02]\end{center}
  \vfill
\end{frame}

\begin{frame}[t]{Our 0D Model vs.\ the 3D Geometric Model}
  Everything in this seminar has a \textbf{full 3D} form and a \textbf{reduced 0D}
  form. We use the \emph{same notation} for both; the badges tell you which:
  \vfill
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \begin{center}\tagd\ \textbf{3D geometric model}\end{center}
      \begin{itemize}\small
        \item full deformation field $\bm{x}=\bm{\varphi}(t,\bm{X})$
        \item tensor $\bm{F}=\partial\bm{\varphi}/\partial\bm{X}$; stresses
        $\bm{\sigma},\bm{S}$
        \item solved on a meshed artery (finite elements)
      \end{itemize}
    \end{column}
    \begin{column}{0.5\textwidth}
      \begin{center}\tagz\ \textbf{0D reduced model}\end{center}
      \begin{itemize}\small
        \item one dominant direction $\Rightarrow$ scalar stretch $\lambda$
        \item tensor $\bm{F}\to\lambda$; stress $\bm{\sigma}\to\sigma$
        \item a handful of ODEs / algebraic equations --- runs in milliseconds
      \end{itemize}
    \end{column}
  \end{columns}
  \vfill
  \begin{center}\small\color{AccentRed}
  The 0D model is the thin-walled artery ($\sigma_\theta=Pa/h$) reduced to its one
  load-bearing direction --- ideal for exploring the theories analytically.
  \end{center}
\end{frame}

\begin{frame}[t]{The Deformation Gradient}
  How a line element transforms is the key kinematic quantity --- the
  \textbf{deformation gradient} maps reference line elements to current ones.
  \dlab{general form --- a tensor}
  \slboxeq{2.4em}{eq:defgrad}{\dd\bm{x} = \bm{F}\,\dd\bm{X},\qquad
    \bm{F}=\frac{\partial\bm{\varphi}}{\partial\bm{X}},\qquad J=\det\bm{F}}
  $J$ is the volume ratio; \textbf{elastic incompressibility} means $J_e=1$ (only
  growth changes volume).
  \zlab{reduction used here --- one dominant direction}
  \giveneq{\bm{F}\ \longrightarrow\ \lambda=\frac{\text{current length}}
    {\text{reference length}},\qquad \lambda=1\ \text{undeformed}.}
\end{frame}

\begin{frame}[t]{Strain --- A Rotation-Free Measure of Stretching}
  $\bm{F}$ still contains rigid rotation. Removing it gives the \textbf{right
  Cauchy--Green} tensor and the \textbf{Green--Lagrange strain}.
  \dlab{general form --- tensors}
  \slboxeq{2.4em}{eq:strain}{\bm{C}=\bm{F}^{\!\top}\bm{F},\qquad
    \bm{E}=\tfrac12\!\left(\bm{C}-\bm{I}\right)}
  \zlab{reduction used here --- one dominant direction}
  \giveneq{C=\lambda^2,\qquad E=\tfrac12(\lambda^2-1).}
  Along the dominant loading direction the whole strain state collapses to a
  single stretch $\lambda$ --- all we need for the thin-walled vessel.
\end{frame}

\begin{frame}[t]{Stress --- Force per Current Area}
  We report the \textbf{Cauchy (true) stress}: force per unit \emph{current} area
  --- the natural measure for a growing body, since the load-bearing area itself
  changes as the tissue remodels.
  \dlab{general form --- a tensor (from the 2nd Piola--Kirchhoff $\bm{S}$)}
  \giveneq{\bm{\sigma}=\tfrac1J\,\bm{F}\,\bm{S}\,\bm{F}^{\!\top},\qquad
    \bm{S}=2\,\frac{\partial W}{\partial\bm{C}}.}
  \zlab{reduction used here --- one scalar}
  \giveneq{\bm{\sigma}\ \longrightarrow\ \sigma\ \text{(the value in Laplace,
    \eqref{eq:laplace})}.}
  Balance of linear momentum in the current configuration reduces, for our
  quasi-static thin-walled vessel, to exactly that Laplace relation.
\end{frame}

\begin{frame}[t]{Hyperelasticity: Stress from a Stored-Energy Function}
  A \textbf{hyperelastic} material stores energy $W$; its stress is a derivative
  of $W$ --- no dissipation, path-independent. Each constituent gets its own $W$:
  \textbf{elastin} is an isotropic \emph{neo-Hookean} solid, \textbf{collagen} a
  \emph{Fung-type} fiber.
  \dlab{general form --- energy $W(\bm{C})$, stress $\bm{S}=2\,\partial W/\partial\bm{C}$}
  \vspace{-0.3em}
  \[
    W_{\text{elastin}}=\tfrac{c}{2}\,(I_1-3),\qquad
    W_{\text{collagen}}=\tfrac{c_1}{4c_2}\!\left[e^{\,c_2(I_4-1)^2}-1\right]
  \]
  \vspace{-0.6em}\par
  {\footnotesize $I_1=\operatorname{tr}\bm{C}$ (isotropic elastin); $I_4$ reads
  $\bm{C}$ along the collagen fiber --- the one place anisotropy enters.}
  \zlab{reduction used here --- one stretch $\lambda_e$, scalar $\sigma=\lambda_e\,\dd W/\dd\lambda_e$}
  \vspace{-0.3em}
  \slboxeq{2.6em}{eq:fung}{\sigma_{\text{elastin}}=c\,(\lambda_e^2-\lambda_e^{-1}),\qquad
    \sigma_{\text{collagen}}=c_1\,\lambda_e^2(\lambda_e^2-1)\,e^{\,c_2(\lambda_e^2-1)^2}}
  \srccite{constituent laws as in Latorre, Humphrey, Comput Methods Appl Mech Eng (2020)}
\end{frame}

\begin{frame}[t]{The Deposition Stretch $G^k$}
  Cells deposit new fibers \textbf{already under tension} at a fixed deposition
  stretch $G^k>1$. At homeostasis the elastic stretch equals $G^k$, which
  \emph{defines} the homeostatic stress:
  \slboxeq{2.4em}{eq:setpoint}{\sigma_h^k := \sigma^k(G^k)}
  \vfill
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \begin{center}\textbf{\color{AccentRed} No deposition pre-stress}\end{center}
      \small New material enters slack ($\sigma_{\text{pre}}=0$); as old fibers are
      removed, the replacement carries no load --- the homeostatic state
      \textbf{cannot be maintained}.
    \end{column}
    \begin{column}{0.5\textwidth}
      \begin{center}\textbf{\color{YaleBlue} With deposition pre-stress}\end{center}
      \small New material enters at the homeostatic prestress
      $\sigma_{\text{pre}}=\sigma_h$ (stretch $G^k$); each replacement immediately
      bears its share --- the state is \textbf{maintained through turnover}.
    \end{column}
  \end{columns}
  \vfill
  \begin{center}\small\color{AccentRed}
  This one convention pins every set-point --- and lets a continuously-renewed
  tissue hold a steady mechanical state.
  \end{center}
  \srccite{deposition-stretch concept: Humphrey, Rajagopal, Math Models Methods Appl Sci (2002)}
\end{frame}
