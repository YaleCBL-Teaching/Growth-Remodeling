\begin{frame}[t]{An Idealized Blood Vessel}
  A vessel is a \textbf{pressurised, perfused tube}. Two loads act on the wall:
  \textbf{blood pressure} $P$ pushes it outward, and \textbf{blood flow} $Q$ drags
  its inner surface. Cells sense both.
  \vfill
  \begin{center}\idealvessel[0.60]\end{center}
  \vfill
  \begin{center}\small
  $a$ inner radius, \ $h$ wall thickness\\
  $P$ lumen pressure $\to$ \textbf{circumferential stress} $\sigma_\theta$\\
  $Q$ volumetric flow $\to$ \textbf{wall shear stress} $\tau_w$\\
  \end{center}
  \srccite{Humphrey, Cardiovascular Solid Mechanics (2002)}
\end{frame}

\begin{frame}[t]{Two Loads, Two Formulae, Two Adaptations}
  Pressure $P$ sets the \textbf{hoop stress}; flow $Q$ sets the \textbf{wall shear
  stress}. Each has a set-point; rearranging shows the geometry that restores it.
  \vfill
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \begin{center}\textbf{\color{YaleBlue} Pressure} (Laplace)\end{center}
      \slboxeq{eq:laplace}{\sigma_\theta = \frac{Pa}{h}}
      % Hold $\sigma_\theta=\sigma_{\theta,h}$: $\;h=Pa/\sigma_{\theta,h}\propto P$ --- higher pressure $\Rightarrow$ \textbf{thicker wall}.
    \end{column}
    \begin{column}{0.5\textwidth}
      \begin{center}\textbf{\color{YaleBlue} Flow} (Poiseuille)\end{center}
      \slboxeq{eq:wss}{\tau_w = \frac{4\mu Q}{\pi\,a^3}}
      % Hold $\tau_w=\tau_{w,h}$: $\;a=(4\mu Q/\pi\tau_{w,h})^{1/3}\propto Q^{1/3}$ --- higher flow $\Rightarrow$ \textbf{larger lumen}.
    \end{column}
  \end{columns}
  \vfill
  \srccite{Laplace \& Poiseuille; Humphrey, Cardiovascular Solid Mechanics (2002)}
\end{frame}

\begin{frame}[t]{}
  \begin{yourturnbox}{Pressure adaptation}
    \qq{Pressure \textbf{doubles}. How does the wall thickness change?
    \quad$\displaystyle\sigma_\theta = \frac{Pa}{h}$}
  \end{yourturnbox}
  \vspace{0.6em}
  \srev
  \begin{answerbox}{Pressure adaptation}
    \begin{minipage}[c]{0.53\linewidth}
      $P'/h'=P/h$ with $P'=2P$ gives
      \slboxeq{eq:ans:pressure}{h' = \left(\frac{P'}{P}\right) h = 2\,h.}
      Wall thickens \emph{in proportion} to pressure.
    \end{minipage}\hfill
    \begin{minipage}[c]{0.45\linewidth}
      \centering
      \includegraphics[height=2.0cm]{figures/p_lo.jpeg}\hspace{0.4em}%
      \includegraphics[height=2.0cm]{figures/p_hi.jpeg}\\[0.15em]
      {\large\color{YaleBlue}$P\uparrow$}
    \end{minipage}
  \end{answerbox}
  \srccite{Hu, \ldots, Humphrey, Ann Biomed Eng (2006)}
\end{frame}

\begin{frame}[t]{}
  \begin{yourturnbox}{Flow adaptation}
    \qq{Flow rises \textbf{8-fold}. How does the radius change?
    \quad$\displaystyle\tau_w = \frac{4\mu Q}{\pi\,a^3}$}
  \end{yourturnbox}
  \vspace{0.6em}
  \srev
  \begin{answerbox}{Flow adaptation}
    \begin{minipage}[c]{0.50\linewidth}
      $(a'/a)^3 = Q'/Q = 8$, so
      \slboxeq{eq:ans:flow}{a' = \left(\frac{Q'}{Q}\right)^{1/3} a = 2\,a.}
      {\color{ABlue}\bfseries Reading.} Radius grows as the \emph{cube root} of flow.
    \end{minipage}\hfill
    \begin{minipage}[c]{0.48\linewidth}
      \centering
      \includegraphics[height=1.7cm]{figures/q_lo.png}\hspace{0.4em}%
      \includegraphics[height=1.7cm]{figures/q_hi.png}\\[0.15em]
      {\large\color{YaleBlue}$Q\uparrow$}
    \end{minipage}
  \end{answerbox}
  \srccite{Rudic, \ldots, Sessa, J Clin Invest (1998)}
\end{frame}
