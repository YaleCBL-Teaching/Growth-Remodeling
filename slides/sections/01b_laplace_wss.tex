\begin{frame}[t]{An Idealized Blood Vessel}
  A vessel is a \textbf{pressurised, perfused tube}. Two loads act on the wall:
  \textbf{blood pressure} $P$ pushes it outward, and \textbf{blood flow} $Q$ drags
  its inner surface. Cells sense both.
  \vfill
  \begin{center}\idealvessel[0.60]\end{center}
  \vfill
  \begin{center}\small
  $P$ lumen pressure $\to$ \textbf{circumferential stress} $\sigma_\theta$
  \qquad
  $Q$ volumetric flow $\to$ \textbf{wall shear stress} $\tau_w$
  \qquad
  $a$ inner radius, \ $h$ wall thickness
  \end{center}
  \srccite{Humphrey, Cardiovascular Solid Mechanics (2002)}
\end{frame}

\begin{frame}[t]{Two Loads, Two Formulae: Laplace and Poiseuille}
  Balancing lumen pressure against wall tension (thin-walled tube) gives the
  \textbf{circumferential (hoop) stress}; steady laminar (Poiseuille) flow gives the
  \textbf{wall shear stress}:
  \vfill
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \begin{center}\textbf{\color{YaleBlue} Pressure load}\end{center}
      \slboxeq{2.6em}{eq:laplace}{\sigma_\theta = P\,\frac{a}{h}}
      Thicker wall $\Rightarrow$ lower stress; larger radius or pressure
      $\Rightarrow$ higher stress. The load the intramural cells (SMC, fibroblasts) feel.
    \end{column}
    \begin{column}{0.5\textwidth}
      \begin{center}\textbf{\color{YaleBlue} Flow load}\end{center}
      \slboxeq{2.6em}{eq:wss}{\tau_w = Q\,\frac{4\mu}{\pi\,a^3}}
      The endothelium senses $\tau_w$; note the steep $a^{-3}$ dependence --- a small
      radius change moves shear stress a lot.
    \end{column}
  \end{columns}
  \vfill
  \srccite{Laplace \& Poiseuille; Humphrey, Cardiovascular Solid Mechanics (2002)}
\end{frame}

\begin{frame}[t]{Two Homeostatic Set-Points, Two Adaptations}
  Each stimulus has a preferred value the cells defend --- the \emph{set-points}
  $\sigma_{\theta,h}$ and $\tau_{w,h}$. Rearranging the two laws shows \emph{what
  geometry restores each set-point}:
  \vfill
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \begin{center}\textbf{\color{YaleBlue} Pressure $\to$ thickness}\end{center}
      Hold $\sigma_\theta=\sigma_{\theta,h}$:
      \giveneq{h = \frac{P\,a}{\sigma_{\theta,h}}\ \propto\ P.}
      Higher pressure $\Rightarrow$ \textbf{thicker wall}.
    \end{column}
    \begin{column}{0.5\textwidth}
      \begin{center}\textbf{\color{YaleBlue} Flow $\to$ radius}\end{center}
      Hold $\tau_w=\tau_{w,h}$:
      \giveneq{a = \left(\frac{4\mu\,Q}{\pi\,\tau_{w,h}}\right)^{1/3}\!\!\propto\ Q^{1/3}.}
      Higher flow $\Rightarrow$ \textbf{larger lumen}.
    \end{column}
  \end{columns}
  \vfill
  \begin{center}\small\color{AccentRed}
  These two rules are the mechanobiology in miniature --- and we can explore them
  analytically before writing any code.
  \end{center}
  \srccite{Humphrey, Cardiovascular Solid Mechanics (2002)}
\end{frame}

% ---- interactive analytical exercises (pen & paper) ------------------------
\begin{frame}[t]{}
  \begin{yourturnbox}{Pressure adaptation}
    \qq{A patient becomes hypertensive: pressure rises by \textbf{50\%} to
    $P' = 1.5\,P$, but the radius $a$ is held fixed. By how much must the wall
    thicken to bring the circumferential stress back to its set-point
    $\sigma_{\theta,h}$? \qkind{pen \& paper}}
    \qres{The Laplace law \eqref{eq:laplace}; homeostatic stress $\sigma_{\theta,h}$.}
    \qhint{Write $\sigma_\theta$ before and after; set the after-value equal to
    $\sigma_{\theta,h}$ and solve for the new thickness $h'$.}
  \end{yourturnbox}
\end{frame}

\begin{frame}[t]{}
  \begin{answerbox}{Pressure adaptation}
    Initially homeostatic: $\sigma_{\theta,h} = P a/h$. After the step, demanding
    $\sigma_\theta = P' a / h' = \sigma_{\theta,h}$ with $P'=1.5\,P$ gives
    \slboxeq{2.2em}{eq:ans:pressure}{h' = 1.5\,h\quad(\text{a }50\%\text{ rise}).}
    \par\smallskip
    {\color{ABlue}\bfseries Reading.} The wall thickens \emph{in proportion} to the
    pressure rise --- exactly Grossman's observation, and the outcome a healthy
    hypertensive adaptation should reproduce. Radius unchanged, stress restored.
  \end{answerbox}
  \srccite{Grossman, Jones, McLaurin, J Clin Invest (1975)}
\end{frame}

\begin{frame}[t]{}
  \begin{yourturnbox}{Flow adaptation}
    \qq{Chronically increased flow rises by \textbf{30\%} to $Q' = 1.3\,Q$. By how
    much must the lumen radius change to restore the wall-shear-stress set-point
    $\tau_{w,h}$? \qkind{pen \& paper}}
    \qres{The wall-shear-stress law \eqref{eq:wss}; set-point $\tau_{w,h}$.}
    \qhint{Only $a$ adapts. Set $\tau_w(Q',a') = \tau_{w,h}$ and use
    $\tau_{w,h} = 4\mu Q/(\pi a^3)$ to eliminate constants.}
  \end{yourturnbox}
\end{frame}

\begin{frame}[t]{}
  \begin{answerbox}{Flow adaptation}
    From $\tau_{w,h} = 4\mu Q'/(\pi a'^3) = 4\mu Q/(\pi a^3)$, the radius ratio is
    $(a'/a)^3 = Q'/Q = 1.3$, so
    \slboxeq{2.2em}{eq:ans:flow}{a' = 1.3^{1/3}\,a \approx 1.09\,a\quad(\text{a }9\%\text{ rise}).}
    \par\smallskip
    {\color{ABlue}\bfseries Reading.} The lumen enlarges as the \emph{cube root} of
    the flow increase --- the classic flow-induced outward remodeling. A $30\%$
    flow rise widens the vessel by only $\sim$9\%.
  \end{answerbox}
  \srccite{Kamiya \& Togawa, Am J Physiol (1980); Humphrey, Cardiovascular Solid Mechanics (2002)}
\end{frame}
