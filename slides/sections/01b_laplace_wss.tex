\begin{frame}[t]{An Idealized Blood Vessel}
  A vessel is a \textbf{pressurised, perfused tube}. Two loads act on the wall:
  \textbf{blood pressure} $P$ pushes it outward, and \textbf{blood flow} $Q$ drags
  its inner surface. Cells sense both.
  \vspace{0.2em}
  \begin{center}\idealvessel[0.58]\end{center}
  \vspace{0.1em}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \begin{itemize}\small
        \item[$P$] lumen pressure $\to$ \textbf{circumferential stress}
        $\sigma_\theta$
        \item[$Q$] volumetric flow $\to$ \textbf{wall shear stress} $\tau_w$
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{itemize}\small
        \item[$a$] inner radius, \ $h$ wall thickness
      \end{itemize}
      {\color{AccentRed}\small Cells regulate \emph{both} $\sigma_\theta$ and
      $\tau_w$ --- by changing $h$ and $a$.}
    \end{column}
  \end{columns}
  \srccite{Humphrey, Cardiovascular Solid Mechanics (2002)}
\end{frame}

\begin{frame}[t]{Circumferential Stress --- The Laplace Law}
  Balancing lumen pressure against wall tension for a thin-walled tube gives the
  \textbf{circumferential (hoop) stress}:
  \slboxeq{2.6em}{eq:laplace}{\sigma_\theta = \frac{P\,a}{h}}
  Thicker wall $\Rightarrow$ lower stress; larger radius or pressure
  $\Rightarrow$ higher stress. This is the load the intramural cells (SMC,
  fibroblasts) feel.
  \srccite{Laplace law; Humphrey, Cardiovascular Solid Mechanics (2002)}
\end{frame}

\begin{frame}[t]{Wall Shear Stress --- The Flow the Endothelium Feels}
  For steady laminar (Poiseuille) flow $Q$ of a fluid with viscosity $\mu$ through
  a tube of radius $a$, the shear stress the flowing blood exerts on the wall is
  \slboxeq{2.6em}{eq:wss}{\tau_w = \frac{4\mu\,Q}{\pi\,a^3}}
  The endothelium senses $\tau_w$; note the steep $a^{-3}$ dependence --- a small
  radius change moves shear stress a lot.
  \srccite{Poiseuille flow; Humphrey, Cardiovascular Solid Mechanics (2002)}
\end{frame}

\begin{frame}[t]{Two Homeostatic Set-Points, Two Adaptations}
  Each stimulus has a preferred value the cells defend. Rearranging the two laws
  shows \emph{what geometry restores each set-point}:
  \vfill
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \begin{center}\textbf{\color{YaleBlue} Pressure $\to$ thickness}\end{center}
      Hold $\sigma_\theta=\sigma_h$:
      \giveneq{h = \frac{P\,a}{\sigma_h}\ \propto\ P.}
      Higher pressure $\Rightarrow$ \textbf{thicker wall}.
    \end{column}
    \begin{column}{0.5\textwidth}
      \begin{center}\textbf{\color{YaleBlue} Flow $\to$ radius}\end{center}
      Hold $\tau_w=\tau_h$:
      \giveneq{a = \left(\frac{4\mu\,Q}{\pi\,\tau_h}\right)^{1/3}\!\!\propto\ Q^{1/3}.}
      Higher flow $\Rightarrow$ \textbf{larger lumen}.
    \end{column}
  \end{columns}
  \vfill
  \begin{center}\small\color{AccentRed}
  These two rules are the mechanobiology in miniature --- and we can explore them
  analytically before writing any code.
  \end{center}
  \srccite{Humphrey, Cardiovascular Solid Mechanics (2002)}
\end{frame}

% ---- interactive analytical exercises (pen & paper) ------------------------
\begin{frame}[t]{}
  \begin{yourturnbox}{Pressure adaptation}
    \qq{A patient becomes hypertensive: pressure rises to $P' = (1+\gamma)P$, but
    the radius $a$ is held fixed. By how much must the wall thicken to bring the
    circumferential stress back to its set-point $\sigma_h$? \qkind{pen \& paper}}
    \qres{The Laplace law \eqref{eq:laplace}; homeostatic stress $\sigma_h$.}
    \qhint{Write $\sigma_\theta$ before and after; set the after-value equal to
    $\sigma_h$ and solve for the new thickness $h'$.}
  \end{yourturnbox}
\end{frame}

\begin{frame}[t]{}
  \begin{answerbox}{Pressure adaptation}
    Initially homeostatic: $\sigma_h = P a/h$. After the step, demanding
    $\sigma_\theta = P' a / h' = \sigma_h$ with $P'=(1+\gamma)P$ gives
    \slboxeq{2.2em}{eq:ans:pressure}{h' = (1+\gamma)\,h.}
    \par\smallskip
    {\color{ABlue}\bfseries Reading.} The wall thickens \emph{in proportion} to the
    pressure rise --- exactly Grossman's observation, and the outcome a healthy
    hypertensive adaptation should reproduce. Radius unchanged, stress restored.
  \end{answerbox}
  \srccite{Grossman, Jones, McLaurin, J Clin Invest (1975)}
\end{frame}

\begin{frame}[t]{}
  \begin{yourturnbox}{Flow adaptation}
    \qq{Chronically increased flow raises $Q' = (1+\varepsilon)Q$. By how much must
    the lumen radius change to restore the wall-shear-stress set-point $\tau_h$?
    \qkind{pen \& paper}}
    \qres{The wall-shear-stress law \eqref{eq:wss}; set-point $\tau_h$.}
    \qhint{Only $a$ adapts. Set $\tau_w(Q',a') = \tau_h$ and use
    $\tau_h = 4\mu Q/(\pi a^3)$ to eliminate constants.}
  \end{yourturnbox}
\end{frame}

\begin{frame}[t]{}
  \begin{answerbox}{Flow adaptation}
    From $\tau_h = 4\mu Q'/(\pi a'^3) = 4\mu Q/(\pi a^3)$, the radius ratio is
    $(a'/a)^3 = Q'/Q = 1+\varepsilon$, so
    \slboxeq{2.2em}{eq:ans:flow}{a' = (1+\varepsilon)^{1/3}\,a.}
    \par\smallskip
    {\color{ABlue}\bfseries Reading.} The lumen enlarges as the \emph{cube root} of
    the flow increase --- the classic flow-induced outward remodeling. A $\sim$30\%
    flow rise widens the vessel by only $\sim$9\%.
  \end{answerbox}
  \srccite{Kamiya \& Togawa, Am J Physiol (1980); Humphrey, Cardiovascular Solid Mechanics (2002)}
\end{frame}
