\begin{frame}[t]{}
  \vfill\begin{center}{\Huge\color{YaleBlue} Homogenized CMM}\\[0.5em]
  {\color{BoxGray} Cyron, Aydin \& Humphrey (2016)}\end{center}\vfill
\end{frame}

\begin{frame}[t]{Throw away the history, keep its average}
  \textbf{Temporal homogenization} replaces the entire deposition history by a
  \emph{single evolving mean natural configuration}. The heredity integral becomes
  \textbf{two ODEs per constituent}; cost drops from $O(N^2)$ to $O(N)$:
  \giveneq{\lambda_e^k = \frac{\lambda}{\lambda_n^k}.}
  \bigfigw{homog_configs.png}{0.88\linewidth}{One mean natural configuration per
  constituent, evolving from reference through initial to current state.}
  \figcredit{Fig.: A. Gebauer (TUM lecture); Cyron, Aydin, Humphrey, Biomech Model Mechanobiol (2016)}
\end{frame}

\begin{frame}[t]{Growth (mass) and remodeling (natural configuration)}
  \textbf{Growth} --- mass production $\propto$ current mass, driven by the
  tissue-stress deviation:
  \slboxeq{2.4em}{eq:homog_mass}{\frac{\dd M^k}{\dd t}
    = M^k\,(K_\sigma^k k_d^k)\left(\frac{\bar\sigma}{\bar\sigma_h}-1\right)}
  Since production scales with current mass, $\dot M^k=0 \Leftrightarrow
  \bar\sigma=\bar\sigma_h$ for \emph{any} mass --- mass settles wherever it must to
  restore tissue stress.
  \vspace{0.3em}

  \textbf{Remodeling} --- new mass enters at $\lambda/G^k$, old mass leaves at the
  current $\lambda_n^k$; mixing at the turnover rate,
  \slboxeq{2.4em}{eq:homog_remodel}{\frac{\dd\lambda_n^k}{\dd t}
    = \frac{m^k}{M^k}\left(\frac{\lambda}{G^k}-\lambda_n^k\right)}
  Holding $\lambda$ fixed, this relaxes constituent stress exponentially to
  $\sigma_h^k$ (time constant $\sim 1/k_d$) --- Cyron's \textbf{mechanical analog}:
  a Maxwell element in parallel with a motor holding the homeostatic prestress.
  \srccite{Cyron, Aydin, Humphrey, Biomech Model Mechanobiol (2016)}
\end{frame}

\begin{frame}[t]{It tracks the full model closely --- much cheaper}
  Two ODEs per constituent replace the full deposition history. The homogenized
  trajectory sits almost on the full constrained mixture, while kinematic growth
  under-responds to elastin loss --- it grows constituents in lockstep and cannot
  remodel their natural configurations individually.
  \vfill
  \begin{center}\color{AccentRed}
  The homogenized model follows the full model's \emph{time trajectory} closely,
  at $O(N)$ instead of $O(N^2)$.
  \end{center}
  \vfill
  {\footnotesize\color{BoxGray} You will overlay the two models and measure the gap
  in the exercise next.}
\end{frame}
