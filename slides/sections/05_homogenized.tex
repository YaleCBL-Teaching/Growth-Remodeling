\begin{frame}[t]{Homogenize: One Mean Natural Configuration per Constituent}
  The full model tracks \emph{every} past cohort with its own natural
  configuration ($O(N^2)$). \textbf{Temporal homogenization} instead gives each
  constituent a \emph{single, evolving} mean natural configuration --- its whole
  mass shares one homogenized elastic deformation, so two ODEs per constituent
  replace the entire history ($O(N)$).
  \dlab{general form (Cyron 2016) --- homogenized elastic deformation}
  \giveneq{\bm{F}^{\alpha}_{e}(s) = \bm{F}(s)\,\big[\bm{F}^{\alpha}_{n}(s)\big]^{-1}}
  \zlab{0D reduction --- scalar mean natural stretch}
  \giveneq{\lambda_e^{\alpha} = \lambda/\lambda_n^{\alpha}}
  \begin{center}\homogmap[0.56]\end{center}
  \srccite{Cyron, Aydin, Humphrey, Biomech Model Mechanobiol (2016)}
\end{frame}

% ---- Maxwell analog moved EARLY: the visual intuition for the whole model ----
\begin{frame}[t]{The Mechanical Analog: A Maxwell Fluid with a Motor}
  Each constituent behaves like a \textbf{Maxwell element in parallel with a
  motor}: turnover relaxes stress toward a \emph{prestress}, not zero.
  \begin{center}\maxwellelement[0.78]\end{center}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \footnotesize
      \renewcommand{\arraystretch}{1.5}
      \begin{tabular}{@{}ll@{}}
        $\sigma^{\alpha}$ & constituent stress (the load)\\[2pt]
        $\mathbb{C}^{\alpha}$ & elastic stiffness (spring)\\[2pt]
        $\dot\varrho_0^{\alpha}/\varrho_0^{\alpha}+1/T^{\alpha}$ & turnover + degradation (dashpot)\\[2pt]
        $\bar\sigma^{\alpha}_{\text{pre}}$ & deposition prestress (motor)\\
      \end{tabular}
    \end{column}
    \begin{column}{0.48\textwidth}
      \footnotesize \textbf{Back to the full CMM (0D):}
      \begin{itemize}\itemsep3pt
        \item spring $\leftrightarrow$ constituent law $\sigma^{\alpha}(\lambda_e^{\alpha})$
        \item dashpot $\leftrightarrow$ mass turnover / removal $k_d^{\alpha}$
        \item motor $\leftrightarrow$ deposition stretch $G^{\alpha}$
      \end{itemize}
    \end{column}
  \end{columns}
  \srccite{Cyron, Aydin, Humphrey, Biomech Model Mechanobiol (2016)}
\end{frame}

\begin{frame}[t]{Growth (Mass) and Remodeling (Natural Configuration)}
  Two 0D ODEs per constituent replace the cohort history. \textbf{Growth}: mass
  production $\propto$ mass, driven by the tissue-stress deviation. \textbf{Remodeling}:
  new mass enters at natural stretch $\lambda/G^{\alpha}$, old mass leaves at the
  current $\lambda_n^{\alpha}$; mixing the two at the turnover rate evolves the mean.
  \begin{columns}[c]
    \begin{column}{0.5\textwidth}
      \zlab{growth --- mass}
      \slboxeq{2.2em}{eq:homog_mass}{\frac{\dd M^{\alpha}}{\dd t}
        = M^{\alpha} K_\sigma^{\alpha} k_d^{\alpha}\!\left(\tfrac{\bar\sigma}{\bar\sigma_h}-1\right)}
    \end{column}
    \begin{column}{0.5\textwidth}
      \zlab{remodeling --- mean natural stretch}
      \slboxeq{2.2em}{eq:homog_remodel}{\frac{\dd\lambda_n^{\alpha}}{\dd t}
        = \frac{m^{\alpha}}{M^{\alpha}}\!\left(\frac{\lambda}{G^{\alpha}}-\lambda_n^{\alpha}\right)}
    \end{column}
  \end{columns}
  \vspace{0.3em}
  with production $m^{\alpha}=k_d^{\alpha}M^{\alpha}\Upsilon^{\alpha}$. Holding
  $\lambda$ fixed, remodeling relaxes each constituent's stress to its prestress
  $\sigma_h^{\alpha}$ (time constant $\sim\!1/k_d$) --- the Maxwell-with-motor
  behaviour.
  \srccite{Cyron, Aydin, Humphrey, Biomech Model Mechanobiol (2016)}
\end{frame}

\begin{frame}[t]{}
  \begin{yourturnbox}{Quick Check --- Why Is the Homogenized Model Cheap?}
    \qq{Its entire state is $(M^{\alpha},\lambda_n^{\alpha})$ per constituent --- a
    \emph{fixed} number of values, updated once per step. How does its cost scale
    with the number of steps $N$, compared with the full model's $O(N^2)$?}
    \qhint{No growing history to re-sum: constant work per step.}
  \end{yourturnbox}
  \vfill
  \begin{center}
  homogenized: $O(N)$ \qquad vs.\qquad full CMM: $O(N^2)$
  \end{center}
  \vfill
  \begin{center}\small
  Same time trajectory, far cheaper --- and the fixed state maps onto a
  viscoelastic (Maxwell-with-motor) material, so it drops into standard
  finite-element solvers.
  \end{center}
\end{frame}

% ---- pen & paper: when are the two models close? ---------------------------
\begin{frame}[t]{}
  \begin{yourturnbox}{When do homogenized and full agree?}
    \qq{The homogenized model keeps only the \emph{mean} natural configuration per
    constituent. Under what assumptions should it match the full deposition-history
    model --- and when would you expect them to diverge? \qkind{discuss}}
    \qres{The two remodeling pictures: full heredity integral vs.\ the single mean
    stretch $\lambda_n^{\alpha}$ \eqref{eq:homog_remodel}.}
    \qhint{Think about the \emph{spread} of natural configurations across cohorts.
    When is a single mean a good summary of the whole history?}
  \end{yourturnbox}
\end{frame}

\begin{frame}[t]{}
  \begin{answerbox}{When do homogenized and full agree?}
    They agree when each constituent's cohorts share \emph{nearly the same} natural
    configuration --- i.e.\ slow, gradual loading where deposition stretch $G^{\alpha}$ and
    turnover are roughly stationary, so the cohort spread stays narrow and the mean
    represents it well. {\color{ABlue}\bfseries They diverge} under fast or strongly
    time-varying insults, when cohorts deposited at very different states coexist and
    a single mean stretch can no longer capture the distribution.
  \end{answerbox}
\end{frame}
