% ===========================================================================
%  Idealized blood vessel drawn in TikZ (thick-walled cylinder cross-section)
%    \idealvessel        -> full annotated version (P, Q, r, h, sigma_theta, tau_w)
% ============================================================================
% Idealized vessel as a LONGITUDINAL CUT (pipe sliced along its axis):
%   radial quantities (a, h, P) read vertically; axial (Q, tau_w) horizontally.
% ---- homeostatic G&R feedback loop (TikZ) \gnrloop -------------------------
% Applied hemodynamic load -> biomechanical state (regulated sigma, tau_w) ->
% homeostatic checks (set-points vs stimuli) -> evolving geometry/composition
% -> single return arrow back into the applied load. (Redraw of the slide-6
% schematic cited to Humphrey, J Elasticity (2021); left-hand fluid/FSI/solid
% detail and the this-work/prior-work split omitted per feedback.)
\newcommand{\gnrloop}[1][1.0]{%
\begin{tikzpicture}[scale=#1,>={Latex[length=2.2mm]},
    box/.style={draw=YaleBlue,very thick,rounded corners=2pt,fill=YaleLightBlue!12,
      align=center,inner sep=6pt,font=\small,text width=6.4cm,minimum height=0.95cm},
    flow/.style={-{Latex[length=2.4mm]},very thick,black!75}]
  \node[box] (load)  at (0,4.5)  {$\mathcal{F}$: applied hemodynamic load\\ \footnotesize (pressure $P$, flow $Q$)};
  \node[box] (state) at (0,3.0)  {biomechanical state\\ \footnotesize regulated variables $\sigma_\theta,\ \tau_w$};
  \node[box] (check) at (0,1.5)  {homeostatic checks\\ \footnotesize set-points vs.\ stimuli: $\Delta\sigma_\theta,\ \Delta\tau_w$};
  \node[box] (evolve) at (0,0.0) {$\mathcal{S}$: evolving geometry,\\ composition, properties};
  \draw[flow] (load)  -- (state);
  \draw[flow] (state) -- (check);
  \draw[flow] (check) -- (evolve);
  % single return arrow: adaptation changes the load the tissue sees
  \draw[flow] (evolve.east) -- ++(4.2,0) |- (load.east);
\end{tikzpicture}}

% ============================================================================
%  Immuno-mechanical regulation: a control-systems block diagram (TikZ)
%  Mechanical stimuli -> state -> sensors; sensors & set-points feed a
%  comparator that emits deviations; the system/model (gains & rates) drives
%  biological responses, closed by a homeostatic (negative) feedback loop and
%  an inflammatory (positive, grey) feedback that resets set-points and gains.
%  Redrawn after Humphrey & Schwartz, Annu. Rev. Biomed. Eng. (2021).
% ============================================================================
\newcommand{\immunoreg}[1][1.0]{%
\begin{tikzpicture}[scale=#1,>={Latex[length=2.4mm]},
    circ/.style={circle,draw=black,thick,fill=black!12,minimum size=0.95cm,
                 inner sep=0pt},
    blk/.style={draw=black,thick,fill=black!8,minimum width=2.3cm,
                minimum height=1.8cm,align=center},
    main/.style={-{Latex[length=2.4mm]},line width=1.1pt},
    posfb/.style={-{Latex[length=2.2mm]},line width=0.9pt,color=black!45},
    every node/.style={font=\small}]
  % --- nodes (loop stretched horizontally) ---
  \node[circ] (state) at (0,0) {};
  \node at (0,1.05) {State};
  \node[circ] (sens) at (2.8,0) {};
  \node at (2.8,-1.05) {Sensors};
  \node[circ] (setp) at (3.6,3.6) {};
  \node at (3.6,4.55) {Set-Points};
  % comparator (triangle)
  \coordinate (tb1) at (5.2,-0.6);
  \coordinate (tb2) at (5.2,1.4);
  \coordinate (tap) at (6.6,0.4);
  \draw[thick,fill=black!8] (tb1)--(tb2)--(tap)--cycle;
  % system / model
  \node[blk] (box) at (11.0,0.4) {Gains and\\ Rates};
  \node[align=center] at (11.0,-1.7) {SYSTEM / MODEL\\ min\{deviations\}};
  % biological responses + branch dot
  \coordinate (dot) at (13.8,0.4);
  \node[align=left,anchor=west,font=\bfseries] (bio) at (14.1,0.4)
       {Biological\\ Responses};
  % mechanical stimuli
  \node[align=center,anchor=east] (stim) at (-1.6,0)
       {\textbf{Mechanical}\\[-1pt] \textbf{Stimuli}\\[1pt] (pressure,\\flow,\ldots)};
  % --- main signal path (black) ---
  \draw[main] (stim.east) -- (state.west);
  \draw[main] (state.east) -- (sens.west);
  \draw[main] (sens.east) -- (5.2,0);             % sensors  -> comparator (low)
  \draw[main] (setp.south) |- (5.2,0.75);         % set-points-> comparator (high)
  \draw[main] (tap) -- (box.west);
  \node at (7.6,1.0) {deviations};                 % clear of the arrow and the box
  \draw[line width=1.1pt] (box.east) -- (dot);
  \fill (dot) circle (2.6pt);
  \draw[main] (dot) -- (bio.west);
  % --- homeostatic (negative) feedback ---
  \draw[main] (dot) -- (13.8,-3.3) -- (0,-3.3) -- (state.south);
  \draw[line width=1.1pt] (-0.36,-0.62) -- (0.36,-0.62);   % summing-junction tick
  \node[align=center] at (6.7,-4.0) {Homeostatic Processes\\ (negative feedback)};
  % --- inflammatory (positive) feedback, grey ---
  \draw[posfb] (dot) -- (13.8,5.3) -- (4.5,5.3) -- (4.5,3.6) -- (setp.east);
  \draw[posfb] (11.0,5.3) -- (box.north);
  \node[align=center,color=black!55] at (8.2,6.1)
       {Inflammatory Processes\\ (positive feedback)};
\end{tikzpicture}}

% ---- thin-walled pressurised tube (Wolinsky-Glagov / Laplace schematic) -----
% The \idealvessel longitudinal-cut sketch, stripped to the geometry and loads
% that matter here: lumen pressure P, inner radius a, wall thickness h, and the
% circumferential (hoop) stress sigma_theta.  Flow Q and wall shear tau_w are
% removed.
\newcommand{\wgvessel}[1][1.0]{%
\begin{tikzpicture}[scale=#1,>={Latex[length=1.8mm]}]
  \def\L{6.0}      % axial length
  \def\a{1.6}      % inner radius (axis -> inner wall)
  \def\h{0.7}      % wall thickness
  % --- shaded lumen (blood-filled interior) ---
  \fill[AccentRed!7] (0,-\a) rectangle (\L,\a);
  % --- wall bands (top & bottom) ---
  \fill[YaleLightBlue!22] (0,\a) rectangle (\L,\a+\h);
  \fill[YaleLightBlue!22] (0,-\a) rectangle (\L,-\a-\h);
  \draw[YaleBlue,very thick] (0,\a) rectangle (\L,\a+\h);
  \draw[YaleBlue,very thick] (0,-\a) rectangle (\L,-\a-\h);
  % --- blood pressure P: pushes outward on both inner surfaces ---
  \foreach \x in {2.0,3.0,4.0}{
    \draw[->,YaleLightBlue,thick] (\x,0.55) -- (\x,\a-0.06);
    \draw[->,YaleLightBlue,thick] (\x,-0.55) -- (\x,-\a+0.06);}
  \node[YaleLightBlue] at (3.0,0.0) {\large $P$};
  % --- circumferential (hoop) stress sigma_theta: wraps AROUND the tube, i.e.
  %     acts OUT of this longitudinal-cut plane -> out-of-page markers on the
  %     wall (as in \idealvessel) ---
  \foreach \x in {2.4,3.6,4.8}{\node[YaleBlue] at (\x,\a+\h/2) {$\odot$};}
  \node[YaleBlue,anchor=south] at (3.6,\a+\h+0.10) {$T$ and $\sigma_\theta$};
  % --- radius a (axis -> inner wall) ---
  \draw[<->,thick] (0.75,0) -- (0.75,\a) node[midway,left]{$a$};
  \fill (0.75,0) circle (1.4pt);
  % --- wall thickness h ---
  \draw[<->,thick,AccentRed] (\L+0.28,\a) -- (\L+0.28,\a+\h)
       node[midway,right]{$h$};
\end{tikzpicture}}
